\section*{}

\begin{tad}{\nombretad{SecuOrdSinRep($\alpha$)}}

\vspace*{11pt}

\extiende{Secu($\alpha$)}

\generos{secuOrdSinRep($\alpha$)}

\exporta{genero, generadores, observadores b\'asicos}
	
\alinearfuncs{agregarOrdenado}{\hspace{5cm}}	
	
\generadores
	\func{agregarOrdenado}{$\alpha$/a, secuOrdSinRep($\alpha$)/s}{secuOrdSinRep($\alpha$)}{$\neg$esta?(a,s)}

\vspace*{11pt}

\otrasops
		\func{ordenada}{secuOrdSinRep($\alpha$)}{bool}{}

\vspace*{11pt}

\alinearaxiomas{ordenada(agregarOrdenado(a, s))}{\hspace{3cm}}
\axiomas{($\forall$ s: secu($\alpha$))($\forall$ a: $\alpha$)}

		\axioma{vacia?(agregarOrdenado(a, s))}{\false}
		
		\vspace{11pt}
		
		\axioma{prim(agregarOrdenado(a, s))}{\lif (vacia?(s)) \lthen a \lelse min(a,prim(s))}
		
		\vspace{11pt}

		\axioma{fin(agregarOrdenado(a, s))}{\lif (vacia?(s)) \oluego\ a $<_\alpha$ prim(s)) \lthen s \lelse agregarOrdenado(a, fin(s)))}

		\vspace{11pt}		
		
		\axioma{ordenada($<>$)}{true}
		\axioma{ordenada(agregarOrdenado(a, s))}{(a $<_\alpha$ prim(s))) $\wedge$ ordenada(s))}
		
\end{tad}
